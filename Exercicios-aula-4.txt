DESAFIO:

CREATE TABLE clientes(
	id SERIAL PRIMARY KEY,
	nome VARCHAR(50) NOT NULL
);

CREATE TABLE pedidos(
	id SERIAL PRIMARY KEY,
	nome VARCHAR(50) NOT NULL,
	data_pedido DATE NOT NULL,
	id_cliente INT,
	CONSTRAINT fk_clientes_pedidos
	FOREIGN KEY (id_cliente) REFERENCES clientes(id)
);
